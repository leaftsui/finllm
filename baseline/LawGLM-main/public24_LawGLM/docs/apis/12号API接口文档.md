    """

    例如：

        {"省份": "西藏自治区", "城市": "拉萨市", "区县": "城关区"}
        输出：
        {'省份': '西藏自治区', 
        '城市': '拉萨市', 
        '城市区划代码': '540100000000', 
        '区县': '城关区', 
        '区县区划代码': '540102000000'}
    """
    url = f"https://{DOMAIN}/law_api/s1_b/get_address_code"

        "query_conds": query_conds,
        "need_fields": need_fields
    }
